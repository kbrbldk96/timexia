api_example.js 
// Timexia API örnek kullanım
import { TimexiaClient } from 'timexia-api';

// API istemcisini başlat
const client = new TimexiaClient({
  apiKey: 'YOUR_API_KEY',
  environment: 'production'
});

// Randevuları listele
async function getAppointments() {
  try {
    const appointments = await client.appointments.list({
      start_date: '2023-08-01',
      end_date: '2023-08-31',
      status: 'confirmed'
    });
    
    return appointments;
  } catch (error) {
    console.error('Hata:', error.message);
  }
}
